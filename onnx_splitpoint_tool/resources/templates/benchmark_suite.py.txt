"""
Benchmark harness for an ONNX split benchmark set.

- Runs each case folder (bXXX/) via its generated runner (run_split_onnxruntime.py)
- Collects validation reports (timing + eps-pass + optional agreement KPIs)
- Writes:
  * benchmark_results_<tag>.json  (canonical)
  * benchmark_results_<tag>.csv   (optional convenience)
  * benchmark_summary_<tag>.md    (1-page summary)
  * benchmark_table_<tag>.tex     (paper-ready table)
  * paper_figures_<tag>/...       (paper-ready plots as PDF + SVG)

Notes:
- For ARCS scope (single machine/provider), the default objective is end-to-end latency
  measured by the composed graph (two sessions: part1 -> part2).
- For future work (multi-device pipelining), the bottleneck objective max(part1, part2)
  can be selected via --objective=max_parts.
"""

def _pearson(x: List[float], y: List[float]) -> float:
    if len(x) != len(y) or len(x) < 2:
        return float("nan")
    xm = sum(x) / len(x)
    ym = sum(y) / len(y)
    num = sum((a - xm) * (b - ym) for a, b in zip(x, y))
    denx = math.sqrt(sum((a - xm) ** 2 for a in x))
    deny = math.sqrt(sum((b - ym) ** 2 for b in y))
    den = denx * deny
    return (num / den) if den > 0 else float("nan")


def _spearman(x: List[float], y: List[float]) -> float:
    # Spearman correlation via Pearson on ranks (average ranks for ties).
    def _ranks(vals: List[float]) -> List[float]:
        n = len(vals)
        order = sorted(range(n), key=lambda i: vals[i])
        ranks = [0.0] * n
        i = 0
        # ranks are 1..n
        while i < n:
            j = i
            while j + 1 < n and vals[order[j + 1]] == vals[order[i]]:
                j += 1
            # average rank for ties
            avg = (i + 1 + j + 1) / 2.0
            for k in range(i, j + 1):
                ranks[order[k]] = avg
            i = j + 1
        return ranks

    if len(x) != len(y) or len(x) < 2:
        return float("nan")
    rx = _ranks([float(v) for v in x])
    ry = _ranks([float(v) for v in y])
    return _pearson(rx, ry)


def _kendall_tau(x: List[float], y: List[float]) -> float:
    # simple O(n^2) kendall tau-a (good enough for small K)
    n = len(x)
    if n != len(y) or n < 2:
        return float("nan")
    conc = 0
    disc = 0
    for i in range(n):
        for j in range(i + 1, n):
            dx = x[i] - x[j]
            dy = y[i] - y[j]
            s = dx * dy
            if s > 0:
                conc += 1
            elif s < 0:
                disc += 1
    denom = conc + disc
    return (conc - disc) / denom if denom > 0 else float("nan")


def _linreg(x: List[float], y: List[float]) -> Tuple[float, float, float]:
    # y = a*x + b, returns (a,b,R^2)
    import numpy as np  # type: ignore

    if len(x) != len(y) or len(x) < 2:
        return float("nan"), float("nan"), float("nan")

    X = np.asarray(x, dtype=np.float64).reshape(-1)
    Y = np.asarray(y, dtype=np.float64).reshape(-1)
    mask = np.isfinite(X) & np.isfinite(Y)
    X = X[mask]
    Y = Y[mask]

    if X.size < 2:
        return float("nan"), float("nan"), float("nan")

    A = np.vstack([X, np.ones_like(X)]).T
    try:
        a, b = np.linalg.lstsq(A, Y, rcond=None)[0]
    except np.linalg.LinAlgError:
        return float("nan"), float("nan"), float("nan")

    Yhat = a * X + b
    ss_res = float(np.sum((Y - Yhat) ** 2))
    ss_tot = float(np.sum((Y - float(np.mean(Y))) ** 2))
    r2 = 1.0 - (ss_res / ss_tot) if ss_tot > 0 else float("nan")
    return float(a), float(b), float(r2)



def _fmt(x: Any, nd: int = 3) -> str:
    try:
        if x is None:
            return "-"
        if isinstance(x, bool):
            return "True" if x else "False"
        xf = float(x)
        if math.isnan(xf):
            return "nan"
        return f"{xf:.{nd}f}"
    except Exception:
        return str(x)


def _objective_value(row: Dict[str, Any], objective: str) -> Optional[float]:
    full = row.get("full_mean_ms")
    p1 = row.get("part1_mean_ms")
    p2 = row.get("part2_mean_ms")
    comp = row.get("composed_mean_ms")
    if objective == "full":
        return float(full) if full is not None else None
    if objective == "composed":
        return float(comp) if comp is not None else None
    if objective == "sum_parts":
        if p1 is None or p2 is None:
            return None
        return float(p1) + float(p2)
    if objective == "max_parts":
        if p1 is None or p2 is None:
            return None
        return max(float(p1), float(p2))
    raise ValueError(f"unknown objective: {objective}")

def _write_table_tex(path: Path, rows: List[Dict[str, Any]], tag: str, objective: str, topk: int = 10) -> None:
    def _tex_escape(text: str) -> str:
        # minimal escaping for captions/tt
        return (
            text.replace("\\", "\\textbackslash{}")
                .replace("_", "\\_")
                .replace("%", "\\%")
                .replace("&", "\\&")
                .replace("#", "\\#")
                .replace("$", "\\$")
                .replace("{", "\\{")
                .replace("}", "\\}")
        )

    label_tag = "bench-" + tag.replace("_", "-")
    # top-k by objective (ascending)
    def keyfn(r: Dict[str, Any]) -> float:
        v = _objective_value(r, objective)
        return float("inf") if v is None else float(v)

    rows_sorted = sorted(rows, key=keyfn)[: max(1, min(topk, len(rows)))]

    lines: List[str] = []
    lines.append(r"\begin{table}[t]")
    lines.append(r"\centering")
    lines.append(r"\small")
    lines.append(r"\setlength{\tabcolsep}{4pt}")
    lines.append(r"\begin{tabular}{rrrrrrrr}")
    lines.append(r"\toprule")
    lines.append(r"Boundary & Cut (MiB) & \#T & $F_L$ (GFLOPs) & $F_R$ (GFLOPs) & $t_{\mathrm{full}}$ (ms) & $t_{\mathrm{comp}}$ (ms) & Pass \\")
    lines.append(r"\midrule")
    for r in rows_sorted:
        b = r.get("boundary", "-")
        cut = _fmt(r.get("cut_mib"), 3)
        nt = r.get("cut_tensors", r.get("n_cut_tensors", "-"))
        fl = _fmt(float(r.get("flops_left", 0.0)) / 1e9 if r.get("flops_left") is not None else None, 2)
        fr = _fmt(float(r.get("flops_right", 0.0)) / 1e9 if r.get("flops_right") is not None else None, 2)
        tf = _fmt(r.get("full_mean_ms"), 2)
        tc = _fmt(r.get("composed_mean_ms"), 2)
        ps = r.get("ok", r.get("eps_pass"))
        ps_s = "True" if ps is True else ("False" if ps is False else "-")
        lines.append(f"{b} & {cut} & {nt} & {fl} & {fr} & {tf} & {tc} & {ps_s} \\\\")
    lines.append(r"\bottomrule")
    lines.append(r"\end{tabular}")
    tag_tt = _tex_escape(tag)
    obj_tt = _tex_escape(objective)
    lines.append(r"\\caption{Top split candidates for \\texttt{" + tag_tt + r"} (sorted by objective: \\texttt{" + obj_tt + r"}).}")
    lines.append(r"\label{tab:benchmark_top_" + label_tag + r"}")
    lines.append(r"\end{table}")
    lines.append("")  # trailing newline

    path.parent.mkdir(parents=True, exist_ok=True)
    with path.open("w", encoding="utf-8") as f:
        f.write("\n".join(lines))


def _save_paper_figures(out_dir: Path, rows: List[Dict[str, Any]], tag: str, objective: str, topk: int = 10) -> List[Path]:
    if not HAVE_PLOT or not rows:
        return []
    out_dir.mkdir(parents=True, exist_ok=True)

    # prepare arrays
    flL = [float(r.get("flops_left", float("nan"))) / 1e9 for r in rows]
    flR = [float(r.get("flops_right", float("nan"))) / 1e9 for r in rows]
    t1 = [float(r.get("part1_mean_ms", float("nan"))) for r in rows]
    t2 = [float(r.get("part2_mean_ms", float("nan"))) for r in rows]
    score = [float(r.get("score", r.get("score_pred", float("nan")))) for r in rows]
    cut = [float(r.get("cut_mib", float("nan"))) for r in rows]
    comp = [float(r.get("composed_mean_ms", float("nan"))) for r in rows]
    full = [float(r.get("full_mean_ms", float("nan"))) for r in rows]
    sum_parts = [float(r.get("sum_parts_ms", float("nan"))) for r in rows]
    overhead = [float(r.get("overhead_ms", float("nan"))) for r in rows]

    obj = [float(_objective_value(r, objective) or float("nan")) for r in rows]

    files: List[Path] = []

    # Fig 1: FLOPs vs time (left)
    a, b, r2 = _linreg(flL, t1)
    r = _pearson(flL, t1)
    plt.figure(figsize=(5.2, 4.0))
    plt.scatter(flL, t1)
    xline = [min(flL), max(flL)]
    yline = [a * x + b for x in xline]
    plt.plot(xline, yline)
    plt.xlabel("FLOPs left (GFLOPs)")
    plt.ylabel("Measured time part1 (ms)")
    plt.title(f"Compute model plausibility (left)  r={_fmt(r,2)}, R$^2$={_fmt(r2,2)}")
    p_pdf = out_dir / f"fig_flops_vs_time_left_{tag}.pdf"
    p_svg = out_dir / f"fig_flops_vs_time_left_{tag}.svg"
    plt.tight_layout()
    plt.savefig(p_pdf)
    plt.savefig(p_svg)
    plt.close()
    files += [p_pdf, p_svg]

    # Fig 2: FLOPs vs time (right)
    a, b, r2 = _linreg(flR, t2)
    r = _pearson(flR, t2)
    plt.figure(figsize=(5.2, 4.0))
    plt.scatter(flR, t2)
    xline = [min(flR), max(flR)]
    yline = [a * x + b for x in xline]
    plt.plot(xline, yline)
    plt.xlabel("FLOPs right (GFLOPs)")
    plt.ylabel("Measured time part2 (ms)")
    plt.title(f"Compute model plausibility (right)  r={_fmt(r,2)}, R$^2$={_fmt(r2,2)}")
    p_pdf = out_dir / f"fig_flops_vs_time_right_{tag}.pdf"
    p_svg = out_dir / f"fig_flops_vs_time_right_{tag}.svg"
    plt.tight_layout()
    plt.savefig(p_pdf)
    plt.savefig(p_svg)
    plt.close()
    files += [p_pdf, p_svg]

    # Fig 3: score vs objective time (ranking quality proxy)
    # highlight top-k predicted and top-k measured (objective)
    import numpy as np  # type: ignore

    score_arr = np.asarray(score, dtype=float)
    obj_arr = np.asarray(obj, dtype=float)
    valid = np.isfinite(score_arr) & np.isfinite(obj_arr)
    score_v = score_arr[valid]
    obj_v = obj_arr[valid]

    if len(score_v) >= 2:
        # indices in original list of valid points
        valid_idx = np.where(valid)[0]
        k = max(1, min(topk, len(score_v)))
        pred_order = np.argsort(score_v)  # smaller score => better
        meas_order = np.argsort(obj_v)    # smaller time => better
        pred_top = set(valid_idx[pred_order[:k]].tolist())
        meas_top = set(valid_idx[meas_order[:k]].tolist())

        colors = []
        for i in range(len(rows)):
            if i in pred_top and i in meas_top:
                colors.append("green")  # agree in top-k
            elif i in pred_top:
                colors.append("red")    # predicted top-k only
            elif i in meas_top:
                colors.append("orange") # measured top-k only
            else:
                colors.append(None)

        plt.figure(figsize=(5.6, 4.2))
        for i, r0 in enumerate(rows):
            if not (math.isfinite(score[i]) and math.isfinite(obj[i])):
                continue
            if colors[i] is None:
                plt.scatter(score[i], obj[i], alpha=0.6)
            else:
                plt.scatter(score[i], obj[i], edgecolors="black", linewidths=0.5, s=60, c=colors[i])
        sp = _spearman([float(s) for s in score_v], [float(t) for t in obj_v])
        kt = _kendall_tau([float(s) for s in score_v], [float(t) for t in obj_v])
        plt.xlabel("Predicted score (higher is better)")
        plt.ylabel(f"Measured objective time: {objective} (ms)")
        plt.title(f"Ranking agreement: Spearman={_fmt(sp,2)}, Kendall={_fmt(kt,2)}")
        p_pdf = out_dir / f"fig_score_vs_{objective}_{tag}.pdf"
        p_svg = out_dir / f"fig_score_vs_{objective}_{tag}.svg"
        plt.tight_layout()
        plt.savefig(p_pdf)
        plt.savefig(p_svg)
        plt.close()
        files += [p_pdf, p_svg]

    # Fig 4: noise summary (CV%)
    def cv(mean: float, std: float) -> float:
        return (std / mean) * 100.0 if mean and mean > 0 else float("nan")

    cv_full = [cv(float(r.get("full_mean_ms", float("nan"))), float(r.get("full_std_ms", float("nan")))) for r in rows]
    cv_p1 = [cv(float(r.get("part1_mean_ms", float("nan"))), float(r.get("part1_std_ms", float("nan")))) for r in rows]
    cv_p2 = [cv(float(r.get("part2_mean_ms", float("nan"))), float(r.get("part2_std_ms", float("nan")))) for r in rows]
    cv_comp = [cv(float(r.get("composed_mean_ms", float("nan"))), float(r.get("composed_std_ms", float("nan")))) for r in rows]

    def _mean_std(vals: List[float]) -> Tuple[float, float]:
        vv = [v for v in vals if math.isfinite(v)]
        if not vv:
            return float("nan"), float("nan")
        m = sum(vv) / len(vv)
        s = math.sqrt(sum((v - m) ** 2 for v in vv) / max(1, len(vv) - 1))
        return m, s

    names = ["full", "part1", "part2", "composed"]
    means = []
    stds = []
    for arr in [cv_full, cv_p1, cv_p2, cv_comp]:
        m, s = _mean_std(arr)
        means.append(m)
        stds.append(s)

    plt.figure(figsize=(5.4, 3.6))
    plt.bar(names, means, yerr=stds)
    plt.ylabel("CV (std/mean) [%]")
    plt.title("Measurement noise (across benchmark cases)")
    p_pdf = out_dir / f"fig_noise_cv_{tag}.pdf"
    p_svg = out_dir / f"fig_noise_cv_{tag}.svg"
    plt.tight_layout()
    plt.savefig(p_pdf)
    plt.savefig(p_svg)
    plt.close()
    files += [p_pdf, p_svg]

    # Fig 5 (optional): overhead vs cut size
    if any(math.isfinite(v) for v in overhead) and any(math.isfinite(v) for v in cut):
        plt.figure(figsize=(5.4, 3.8))
        plt.scatter(cut, overhead)
        plt.axhline(0.0)
        plt.xlabel("Cut size (MiB)")
        plt.ylabel("Overhead: composed - (part1+part2) [ms]")
        plt.title("Split overhead vs cut size")
        p_pdf = out_dir / f"fig_overhead_vs_cut_{tag}.pdf"
        p_svg = out_dir / f"fig_overhead_vs_cut_{tag}.svg"
        plt.tight_layout()
        plt.savefig(p_pdf)
        plt.savefig(p_svg)
        plt.close()
        files += [p_pdf, p_svg]

    return files


def _write_figures(out_dir: Path, rows: List[Dict[str, Any]], tag: str, objective: str, topk: int = 10) -> List[Path]:
    """Compatibility alias for older scripts that called `_write_figures`.

    The suite uses `_save_paper_figures` for all run types; `_write_figures` keeps
    older call-sites working.
    """
    return _save_paper_figures(out_dir=out_dir, rows=rows, tag=tag, objective=objective, topk=topk)


def _write_summary_md(path: Path, rows: List[Dict[str, Any]], tag: str, objective: str, topk: int = 10) -> None:
    if not rows:
        path.write_text("# Benchmark summary\n\nNo results.\n", encoding="utf-8")
        return

    # derived objective arrays
    obj = [(_objective_value(r, objective) or float("nan")) for r in rows]
    score = [float(r.get("score", r.get("score_pred", float("nan")))) for r in rows]


    # compute model plausibility (if FLOPs metadata is available)
    def _sf(x):
        try:
            return float(x)
        except Exception:
            return float("nan")

    flL = [_sf(r.get("flops_left")) / 1e9 for r in rows]
    flR = [_sf(r.get("flops_right")) / 1e9 for r in rows]
    t1 = [_sf(r.get("part1_mean_ms")) for r in rows]
    t2 = [_sf(r.get("part2_mean_ms")) for r in rows]

    def _finite_pairs(x: list[float], y: list[float]) -> tuple[list[float], list[float]]:
        out_x: list[float] = []
        out_y: list[float] = []
        for a, b in zip(x, y):
            if math.isfinite(a) and math.isfinite(b):
                out_x.append(a)
                out_y.append(b)
        return out_x, out_y

    flL_v, t1_v = _finite_pairs(flL, t1)
    flR_v, t2_v = _finite_pairs(flR, t2)

    if len(flL_v) >= 2:
        rL = _pearson(flL_v, t1_v)
        aL, bL, r2L = _linreg(flL_v, t1_v)
    else:
        rL = float("nan")
        aL = bL = r2L = float("nan")

    if len(flR_v) >= 2:
        rR = _pearson(flR_v, t2_v)
        aR, bR, r2R = _linreg(flR_v, t2_v)
    else:
        rR = float("nan")
        aR = bR = r2R = float("nan")


    # ranking agreement (score vs objective time)
    import numpy as np  # type: ignore

    score_arr = np.asarray(score, dtype=float)
    obj_arr = np.asarray(obj, dtype=float)
    valid = np.isfinite(score_arr) & np.isfinite(obj_arr)
    score_v = score_arr[valid].tolist()
    obj_v = obj_arr[valid].tolist()
    sp = _spearman([float(s) for s in score_v], [-float(t) for t in obj_v]) if len(score_v) >= 2 else float("nan")
    kt = _kendall_tau([float(s) for s in score_v], [-float(t) for t in obj_v]) if len(score_v) >= 2 else float("nan")

    # top-k overlap (pred score vs measured objective)
    k = max(1, min(topk, len(score_v)))
    pred_order = sorted(range(len(score_v)), key=lambda i: score_v[i], reverse=True)
    meas_order = sorted(range(len(obj_v)), key=lambda i: obj_v[i])
    overlap = len(set(pred_order[:k]).intersection(set(meas_order[:k])))
    overlap_ratio = overlap / k

    # noise summary (CV)
    def cv(mean: float, std: float) -> float:
        return (std / mean) * 100.0 if mean and mean > 0 else float("nan")

    cv_comp = [cv(float(r.get("composed_mean_ms", float("nan"))), float(r.get("composed_std_ms", float("nan")))) for r in rows]
    cv_comp_v = [v for v in cv_comp if math.isfinite(v)]
    cv_comp_med = sorted(cv_comp_v)[len(cv_comp_v) // 2] if cv_comp_v else float("nan")

    def _sf_or_none(value: Any) -> Optional[float]:
        try:
            out = float(value)
        except Exception:
            return None
        return out if math.isfinite(out) else None

    def _median(values: List[Any]) -> float:
        vv = sorted(v for v in (_sf_or_none(x) for x in values) if v is not None)
        if not vv:
            return float("nan")
        mid = len(vv) // 2
        if len(vv) % 2:
            return float(vv[mid])

def _slug(text: str) -> str:
    """Convert an arbitrary id into a filesystem-friendly slug."""
    s = str(text or "").strip()
    if not s:
        return "run"
    out = []
    for ch in s:
        if ch.isalnum() or ch in ("-", "_", "."):
            out.append(ch)
        else:
            out.append("_")
    slug = "".join(out).strip("_")
    while "__" in slug:
        slug = slug.replace("__", "_")
    return slug or "run"


def _flatten_hailo_compile_meta(case_meta: Dict[str, Any]) -> Dict[str, Any]:
    flat: Dict[str, Any] = {}
    payload = case_meta.get("hailo_compile") if isinstance(case_meta, dict) else None
    if not isinstance(payload, dict):
        return flat

    def _ctx_single(meta: Dict[str, Any]) -> bool:
        mode = str(meta.get("context_mode") or "").strip()
        try:
            count = int(float(meta.get("context_count"))) if meta.get("context_count") not in (None, "") else None
        except Exception:
            count = None
        return bool(mode == "single_context_used" or count == 1)

    for hw_arch, hw_meta in payload.items():
        if not isinstance(hw_meta, dict):
            continue
        arch = _slug(str(hw_arch)).lower()
        p1 = hw_meta.get("part1") if isinstance(hw_meta.get("part1"), dict) else {}
        p2 = hw_meta.get("part2") if isinstance(hw_meta.get("part2"), dict) else {}
        for stage, meta in (("part1", p1), ("part2", p2)):
            if not isinstance(meta, dict):
                continue
            for field in (
                "ok",
                "skipped",
                "timed_out",
                "error",
                "failure_kind",
                "unsupported_reason",
                "elapsed_s",
                "context_count",
                "context_mode",
                "partition_iterations",
                "partition_time_s",
                "allocation_time_s",
                "compilation_time_s",
                "calib_source",
                "single_context_failed",
                "single_context_used",
                "multi_context_used",
                "mapping_failed",
                "watchdog_expired",
            ):
                if field in meta:
                    flat[f"hailo_{arch}_{stage}_{field}"] = meta.get(field)
        flat[f"hailo_{arch}_part2_single_context"] = _ctx_single(p2)
        flat[f"hailo_{arch}_both_parts_single_context"] = bool(_ctx_single(p1) and _ctx_single(p2))
    return flat


def _resolve_stage_token(stage: Any) -> Optional[str]:
    """Resolve a stage config dict (from benchmark_plan.json) into a provider token.

    Returns one of:
    - ORT provider token: auto/cpu/cuda/tensorrt
    - Hailo hw_arch token: hailo8/hailo8l/hailo8r/hailo10h/hailo10p
    """
    if not isinstance(stage, dict):
        return None
    ty = str(stage.get("type") or stage.get("kind") or "").strip().lower()
    if ty in {"hailo"}:
        hw = stage.get("hw_arch") or stage.get("arch") or stage.get("id")
        hw = (str(hw).strip().lower() if hw else "")
        return hw or None
    if ty in {"onnxruntime", "ort"}:
        p = stage.get("provider")
        p = (str(p).strip().lower() if p else "")
        return p or None
    return None


def _normalize_variants(values: Any) -> Optional[List[str]]:
    if not isinstance(values, list):
        return None
    out: List[str] = []
    seen: set[str] = set()
    for value in values:
        name = str(value or "").strip().lower()
        if not name or name in seen:
            continue
        seen.add(name)
        out.append(name)
    return out


def _variant_hailo_requirements_for_run(run: Dict[str, Any], variant: str) -> List[Tuple[str, str]]:
    variant_key = str(variant or "").strip().lower()
    stage1_tok = _resolve_stage_token(run.get("stage1"))
    stage2_tok = _resolve_stage_token(run.get("stage2"))
    run_type = str(run.get("type") or "").strip().lower()
    run_hw = str(run.get("hw_arch") or run.get("id") or "").strip().lower() or None

    req: List[Tuple[str, str]] = []
    if variant_key == "full":
        if run_type == "hailo":
            hw = run_hw or stage1_tok or stage2_tok
            if hw:
                req.append((str(hw), "full"))
        return req
    if variant_key == "part1":
        if stage1_tok and stage1_tok.startswith("hailo"):
            req.append((str(stage1_tok), "part1"))
        return req
    if variant_key == "part2":
        if stage2_tok and stage2_tok.startswith("hailo"):
            req.append((str(stage2_tok), "part2"))
        return req
    if variant_key == "composed":
        if stage1_tok and stage1_tok.startswith("hailo"):
            req.append((str(stage1_tok), "part1"))
        if stage2_tok and stage2_tok.startswith("hailo"):
            req.append((str(stage2_tok), "part2"))
        return req
    return req


def _prune_case_variants_for_run(
    run: Dict[str, Any],
    variants: Optional[List[str]],
    case_meta: Optional[Dict[str, Any]],
    suite_hailo_hefs: Optional[Dict[str, Any]] = None,
) -> Optional[List[str]]:
    if variants is None:
        return variants
    availability = case_meta.get("hailo_case_variant_availability") if isinstance(case_meta, dict) else None
    availability = availability if isinstance(availability, dict) else {}
    suite_hailo_hefs = suite_hailo_hefs if isinstance(suite_hailo_hefs, dict) else {}

    out: List[str] = []
    for variant in variants:
        req = _variant_hailo_requirements_for_run(run, variant)
        if not req:
            out.append(variant)
            continue
        ok = True
        for hw, kind in req:
            hw_key = str(hw)
            case_ok = bool((availability.get(hw_key) or {}).get(kind))
            if kind == "full":
                suite_meta = suite_hailo_hefs.get(hw_key) if isinstance(suite_hailo_hefs, dict) else None
                suite_meta = suite_meta if isinstance(suite_meta, dict) else {}
                suite_ok = bool(suite_meta.get("full")) and not bool(suite_meta.get("full_error"))
                ok = ok and bool(case_ok or suite_ok)
            else:
                ok = ok and bool(case_ok)
            if not ok:
                break
        if ok:
            out.append(variant)
    return out

_STATUS_VARIANTS = ["full", "part1", "part2", "composed"]

def _agg_status(values: List[str]) -> str:
    """Aggregate a list of per-candidate statuses into one status.

    Precedence is: ok > error > skipped > missing
    """
    vset = {v for v in values if v}
    if "ok" in vset:
        return "ok"
    if "error" in vset:
        return "error"
    if "skipped" in vset:
        return "skipped"
    return "missing"

def _status_symbol(status: str) -> str:
    return {
        "ok": "✅",
        "error": "❌",
        "skipped": "⏭️",
        "missing": "—",
    }.get(status, status)

def _write_status_matrix(root: Path, planned_runs: List[Dict[str, Any]]) -> None:
    """Write a compact status matrix across all planned runs.

    Uses the `variant_status` dict written by the per-case runner.
    """
    rows_out: List[Dict[str, Any]] = []
    for run in planned_runs:
        tag = run.get("tag") or run.get("id") or ""
        run_id = run.get("id") or tag
        rtype = run.get("type", "?")
        stage1 = _resolve_stage_token(run.get("stage1")) or run.get("provider", "")
        stage2 = _resolve_stage_token(run.get("stage2")) or run.get("provider", "")
        res_path = root / f"benchmark_results_{tag}.json"
        agg: Dict[str, str] = {k: "missing" for k in _STATUS_VARIANTS}
        if res_path.exists():
            try:
                data = _read_json(res_path)
                if isinstance(data, list):
                    for k in _STATUS_VARIANTS:
                        vals: List[str] = []
                        for rec in data:
                            if not isinstance(rec, dict):
                                continue
                            vs = rec.get("variant_status")
                            if isinstance(vs, dict) and k in vs:
                                vals.append(str(vs.get(k) or ""))
                        agg[k] = _agg_status(vals)
            except Exception:
                # Keep defaults if parsing fails
                pass

        row: Dict[str, Any] = {
            "run": str(run_id),
            "tag": str(tag),
            "type": str(rtype),
            "stage1": str(stage1),
            "stage2": str(stage2),
        }
        row.update(agg)
        rows_out.append(row)

    _write_json(root / "benchmark_suite_status_matrix.json", rows_out)

    csv_path = root / "benchmark_suite_status_matrix.csv"
    with csv_path.open("w", newline="", encoding="utf-8") as f:
        w = csv.writer(f)
        w.writerow(["run", "type", "stage1", "stage2"] + _STATUS_VARIANTS)
        for r in rows_out:
            w.writerow([r["run"], r["type"], r["stage1"], r["stage2"]] + [r[k] for k in _STATUS_VARIANTS])

    md_path = root / "benchmark_suite_status_matrix.md"
    lines: List[str] = []
    lines.append("# Benchmark suite status matrix\n")
    lines.append("Legend: ✅ ok, ❌ error, ⏭️ skipped, — missing\n")
    lines.append("| run | type | stage1 | stage2 | full | part1 | part2 | composed |")
    lines.append("|---|---|---|---|---|---|---|---|")
    for r in rows_out:
        sym = {k: _status_symbol(str(r.get(k, "missing"))) for k in _STATUS_VARIANTS}
        lines.append(
            f"| {r['run']} | {r['type']} | {r['stage1']} | {r['stage2']} | {sym['full']} | {sym['part1']} | {sym['part2']} | {sym['composed']} |"
        )
    md_path.write_text("\n".join(lines) + "\n", encoding="utf-8")



def _safe_float(v: Any) -> Optional[float]:
    try:
        if v is None:
            return None
        x = float(v)
        if math.isnan(x) or math.isinf(x):
            return None
        return x
    except Exception:
        return None


def _write_v42_pipeline_summary(root: Path, rows_by_tag: Dict[str, List[Dict[str, Any]]]) -> None:
    """Write a thesis-oriented summary for single-system FPS vs hetero streaming FPS."""
    flat: List[Dict[str, Any]] = []
    full_candidates: List[Dict[str, Any]] = []
    streaming_candidates: List[Dict[str, Any]] = []
    for tag, rows in (rows_by_tag or {}).items():
        for r in rows or []:
            item = {
                "tag": tag,
                "run_id": r.get("run_id"),
                "provider": r.get("provider"),
                "stage1_provider": r.get("stage1_provider"),
                "stage2_provider": r.get("stage2_provider"),
                "case_dir": r.get("case_dir"),
                "boundary": r.get("boundary"),
                "cut_mib": r.get("cut_mib"),
                "n_cut_tensors": r.get("n_cut_tensors"),
                "full_mean_ms": r.get("full_mean_ms"),
                "full_e2e_mean_ms": r.get("full_e2e_mean_ms"),
                "composed_mean_ms": r.get("composed_mean_ms"),
                "throughput_fps_makespan": r.get("throughput_fps_makespan"),
                "throughput_cycle_est_ms": r.get("throughput_cycle_est_ms"),
                "mini_coco_ap50_primary": r.get("mini_coco_ap50_primary"),
                "mini_coco_ap50_full": r.get("mini_coco_ap50_full"),
                "semantic_validation_passed": r.get("semantic_validation_passed"),
                "final_pass_all": r.get("final_pass_all"),
            }
            tag_l = str(tag or "").lower()
            st1_l = str(r.get("stage1_provider") or "").lower()
            st2_l = str(r.get("stage2_provider") or "").lower()
            is_hetero_tag = ("hailo" in st1_l and ("trt" in st2_l or "tensorrt" in st2_l)) or (("trt" in st1_l or "tensorrt" in st1_l) and "hailo" in st2_l) or "_to_" in tag_l
            # Preferred full FPS: fair E2E if present, otherwise legacy full mean.
            # Do not let mixed-run helper rows become the single-system baseline.
            full_ms = _safe_float(r.get("full_e2e_mean_ms")) or _safe_float(r.get("full_mean_ms"))
            if full_ms and full_ms > 0 and not is_hetero_tag:
                cand = dict(item)
                cand["baseline_ms_used"] = full_ms
                cand["baseline_fps"] = 1000.0 / full_ms
                full_candidates.append(cand)
            stream_fps = _safe_float(r.get("throughput_fps_makespan"))
            if stream_fps and stream_fps > 0:
                cand = dict(item)
                cand["streaming_fps"] = stream_fps
                streaming_candidates.append(cand)
            flat.append(item)

    best_full = max(full_candidates, key=lambda x: float(x.get("baseline_fps") or 0.0), default=None)
    best_stream = max(streaming_candidates, key=lambda x: float(x.get("streaming_fps") or 0.0), default=None)
    gain_vs_best_full = None
    if best_full and best_stream and _safe_float(best_full.get("baseline_fps")):
        gain_vs_best_full = (float(best_stream["streaming_fps"]) / float(best_full["baseline_fps"]) - 1.0) * 100.0

    payload = {
        "schema_version": 1,
        "description": "v42 thesis summary: compare single-system full FPS against heterogenous streaming/pipelined split FPS.",
        "best_full_baseline": best_full,
        "best_heterogeneous_streaming": best_stream,
        "gain_vs_best_full_percent": gain_vs_best_full,
        "num_rows": len(flat),
        "num_full_candidates": len(full_candidates),
        "num_streaming_candidates": len(streaming_candidates),
    }
    _write_json(root / "v42_pipeline_summary.json", payload)

    csv_path = root / "v42_pipeline_summary.csv"
    keys = [
        "tag", "run_id", "provider", "stage1_provider", "stage2_provider", "case_dir", "boundary",
        "full_mean_ms", "full_e2e_mean_ms", "composed_mean_ms", "throughput_fps_makespan",
        "throughput_cycle_est_ms", "mini_coco_ap50_primary", "mini_coco_ap50_full",
        "semantic_validation_passed", "final_pass_all",
    ]
    with csv_path.open("w", newline="", encoding="utf-8") as f:
        w = csv.DictWriter(f, fieldnames=keys)
        w.writeheader()
        for row in flat:
            w.writerow({k: row.get(k) for k in keys})

    md_lines = [
        "# v42 Pipeline Summary",
        "",
        "This summary separates single-frame latency from heterogeneous streaming throughput.",
        "",
    ]
    if best_full:
        md_lines.append(f"Best full baseline: `{best_full.get('tag')}` / `{best_full.get('case_dir')}` = {best_full.get('baseline_fps'):.3f} FPS ({best_full.get('baseline_ms_used'):.3f} ms used).")
    else:
        md_lines.append("Best full baseline: not available.")
    if best_stream:
        md_lines.append(f"Best heterogeneous streaming: `{best_stream.get('tag')}` / `{best_stream.get('case_dir')}` = {best_stream.get('streaming_fps'):.3f} FPS.")
    else:
        md_lines.append("Best heterogeneous streaming: not available.")
    if gain_vs_best_full is not None:
        md_lines.append(f"Gain vs best full baseline: **{gain_vs_best_full:.2f}%**.")
    md_lines.append("")
    md_lines.append("| tag | case | full ms | full e2e ms | composed ms | streaming FPS | AP50 primary | semantic pass |")
    md_lines.append("|---|---:|---:|---:|---:|---:|---:|---|")
    def _fmt(v: Any) -> str:
        x = _safe_float(v)
        return "-" if x is None else f"{x:.3f}"
    for row in sorted(flat, key=lambda x: float(_safe_float(x.get("throughput_fps_makespan")) or -1.0), reverse=True)[:25]:
        md_lines.append(
            f"| {row.get('tag')} | {row.get('case_dir')} | {_fmt(row.get('full_mean_ms'))} | {_fmt(row.get('full_e2e_mean_ms'))} | {_fmt(row.get('composed_mean_ms'))} | {_fmt(row.get('throughput_fps_makespan'))} | {_fmt(row.get('mini_coco_ap50_primary'))} | {row.get('semantic_validation_passed')} |"
        )
    (root / "v42_pipeline_summary.md").write_text("\n".join(md_lines) + "\n", encoding="utf-8")

    # v47: explicitly mark the heterogeneous Hailo<->TensorRT streaming claim.
    hetero_streaming = []
    for cand in streaming_candidates:
        tag = str(cand.get("tag") or "").lower()
        st1 = str(cand.get("stage1_provider") or "").lower()
        st2 = str(cand.get("stage2_provider") or "").lower()
        if ("hailo" in st1 and ("trt" in st2 or "tensorrt" in st2)) or (("trt" in st1 or "tensorrt" in st1) and "hailo" in st2) or "hailo8_to_trt" in tag or "trt_to_hailo" in tag:
            hetero_streaming.append(cand)
    best_hetero = max(hetero_streaming, key=lambda x: float(x.get("streaming_fps") or 0.0), default=None)
    gain_hetero_vs_best_full = None
    if best_full and best_hetero and _safe_float(best_full.get("baseline_fps")):
        gain_hetero_vs_best_full = (float(best_hetero["streaming_fps"]) / float(best_full["baseline_fps"]) - 1.0) * 100.0
    v47_payload = dict(payload)
    v47_payload.update({
        "schema_version": 2,
        "description": "v47 thesis summary: single-system full baselines vs heterogeneous Hailo<->TensorRT streaming throughput.",
        "best_heterogeneous_streaming": best_hetero,
        "gain_vs_best_full_percent": gain_hetero_vs_best_full,
        "num_heterogeneous_streaming_candidates": len(hetero_streaming),
    })
    _write_json(root / "v47_pipeline_summary.json", v47_payload)
    v47_csv_path = root / "v47_pipeline_summary.csv"
    with v47_csv_path.open("w", newline="", encoding="utf-8") as f:
        w = csv.DictWriter(f, fieldnames=keys)
        w.writeheader()
        for row in flat:
            w.writerow({k: row.get(k) for k in keys})
    v47_lines = [
        "# v47 Pipeline Summary",
        "",
        "This summary highlights heterogeneous Hailo<->TensorRT streaming throughput separately from single-frame latency.",
        "",
    ]
    if best_full:
        v47_lines.append(f"Best full baseline: `{best_full.get('tag')}` / `{best_full.get('case_dir')}` = {best_full.get('baseline_fps'):.3f} FPS ({best_full.get('baseline_ms_used'):.3f} ms used).")
    else:
        v47_lines.append("Best full baseline: not available.")
    if best_hetero:
        v47_lines.append(f"Best heterogeneous streaming: `{best_hetero.get('tag')}` / `{best_hetero.get('case_dir')}` = {best_hetero.get('streaming_fps'):.3f} FPS.")
    else:
        v47_lines.append("Best heterogeneous streaming: not available.")
    if gain_hetero_vs_best_full is not None:
        v47_lines.append(f"Gain vs best full baseline: **{gain_hetero_vs_best_full:.2f}%**.")
    v47_lines.append("")
    v47_lines.append("| tag | case | full e2e ms | composed ms | streaming FPS | AP50 primary | semantic pass |")
    v47_lines.append("|---|---:|---:|---:|---:|---:|---|")
    for row in sorted(flat, key=lambda x: float(_safe_float(x.get("throughput_fps_makespan")) or -1.0), reverse=True)[:30]:
        v47_lines.append(
            f"| {row.get('tag')} | {row.get('case_dir')} | {_fmt(row.get('full_e2e_mean_ms'))} | {_fmt(row.get('composed_mean_ms'))} | {_fmt(row.get('throughput_fps_makespan'))} | {_fmt(row.get('mini_coco_ap50_primary'))} | {row.get('semantic_validation_passed')} |"
        )
    (root / "v47_pipeline_summary.md").write_text("\n".join(v47_lines) + "\n", encoding="utf-8")


def _collect_case_result(case_dir: Path, out_dir_name: str, run_id: str, provider: str, image_scale: str, stage1: Optional[str] = None, stage2: Optional[str] = None, variants: Optional[List[str]] = None, case_meta: Optional[Dict[str, Any]] = None, *, runner_rc: int = 0) -> Optional[Dict[str, Any]]:
    """Load and normalize an existing per-case validation report.

    This is used both after a freshly executed case and in --resume mode to
    reuse results that are already present on disk.
    """
    report = case_dir / out_dir_name / "validation_report.json"
    if not report.exists():
        report_auto = case_dir / "results_auto" / "validation_report.json"
        report = report_auto if report_auto.exists() else report
    if not report.exists():
        print(f"[warn] missing report: {report}")
        return None

    r = _read_json(report)
    if not isinstance(r, dict):
        print(f"[warn] invalid report payload: {report}")
        return None

    # Attach suite-side metadata
    r.setdefault("run_id", run_id)
    r.setdefault("provider", provider)
    r.setdefault("case_dir", case_dir.name)
    if stage1:
        r.setdefault("stage1_provider", str(stage1))
    if stage2:
        r.setdefault("stage2_provider", str(stage2))
    if variants:
        r.setdefault("requested_variants", list(variants))
    r["_runner_rc"] = runner_rc

    # Flatten predicted fields from newer runner versions (runner may store them under "predicted")
    pred = r.get("predicted")
    if isinstance(pred, dict):
        for k, v in pred.items():
            if k not in r:
                r[k] = v
        # common aliases used by this suite
        if "n_cut_tensors" not in r:
            if "crossing_tensors_all" in pred:
                r["n_cut_tensors"] = pred["crossing_tensors_all"]
            elif "crossing_tensors_known" in pred:
                r["n_cut_tensors"] = pred["crossing_tensors_known"]
            elif isinstance(r.get("cut_tensors"), list):
                r["n_cut_tensors"] = len(r["cut_tensors"])
        if "score" in pred and "score_pred" not in r:
            r["score_pred"] = pred["score"]
        if "imbalance" in pred and "imbalance_pred" not in r:
            r["imbalance_pred"] = pred["imbalance"]
        if "total_flops" in pred and "flops_total" not in r:
            r["flops_total"] = pred["total_flops"]
    # derive total flops if possible
    if "flops_total" not in r and ("flops_left" in r) and ("flops_right" in r):
        try:
            r["flops_total"] = float(r["flops_left"]) + float(r["flops_right"])
        except Exception:
            pass

    # enrich with static case meta if present
    manifest = case_dir / "split_manifest.json"
    if manifest.exists():
        m = _read_json(manifest)
        for k in ["boundary", "cut_mib", "n_cut_tensors", "flops_left", "flops_right", "imbalance_pred", "score_pred"]:
            if k in m and k not in r:
                r[k] = m[k]
        for k, v in m.items():
            r.setdefault(k, v)

    # Merge __BENCH_JSON__ metadata (predicted fields) if provided
    if isinstance(case_meta, dict):
        mp = case_meta.get("predicted")
        if isinstance(mp, dict):
            if not isinstance(r.get("predicted"), dict):
                r["predicted"] = mp
            else:
                for kk, vv in mp.items():
                    r["predicted"].setdefault(kk, vv)
        for kk in ["boundary_index", "boundary_id", "boundary_name", "case_dir", "folder"]:
            if kk in case_meta:
                r.setdefault(kk, case_meta.get(kk))
        for kk, vv in _flatten_hailo_compile_meta(case_meta).items():
            r.setdefault(kk, vv)

    pred2 = r.get("predicted")
    if isinstance(pred2, dict):
        for k, v in pred2.items():
            r.setdefault(k, v)

    if "score_pred" not in r and "score" in r:
        r["score_pred"] = r.get("score")
    if "imbalance_pred" not in r and "imbalance" in r:
        r["imbalance_pred"] = r.get("imbalance")
    if "n_cut_tensors" not in r:
        if "cut_tensors_toggle" in r:
            r["n_cut_tensors"] = r.get("cut_tensors_toggle")
        elif "cut_tensors" in r and isinstance(r.get("cut_tensors"), (int, float)):
            r["n_cut_tensors"] = int(r.get("cut_tensors"))
        elif isinstance(r.get("cut_tensors"), (list, tuple)):
            r["n_cut_tensors"] = len(r.get("cut_tensors"))
        else:
            pred_tmp = r.get("predicted")
            if isinstance(pred_tmp, dict):
                if isinstance(pred_tmp.get("crossing_tensors_all"), (list, tuple)):
                    r["n_cut_tensors"] = len(pred_tmp.get("crossing_tensors_all"))
                elif isinstance(pred_tmp.get("crossing_tensors_known"), (list, tuple)):
                    r["n_cut_tensors"] = len(pred_tmp.get("crossing_tensors_known"))
    if "eps_pass" not in r and "ok" in r:
        r["eps_pass"] = bool(r.get("ok"))

    tms = r.get("timing_ms") or r.get("timing") or r.get("timings")
    if isinstance(tms, dict):
        for _name in ("full", "part1", "part2", "composed"):
            blk = tms.get(_name)
            if isinstance(blk, dict):
                mean = blk.get("mean", blk.get("mean_ms"))
                std = blk.get("std", blk.get("std_ms"))
                if mean is not None:
                    r.setdefault(f"{_name}_mean_ms", mean)
                if std is not None:
                    r.setdefault(f"{_name}_std_ms", std)


    # v42 flattened timing diagnostics.
    if isinstance(tms, dict):
        full_blk = tms.get("full") if isinstance(tms.get("full"), dict) else {}
        part2_blk = tms.get("part2") if isinstance(tms.get("part2"), dict) else {}
        comp_blk = tms.get("composed") if isinstance(tms.get("composed"), dict) else {}
        if isinstance(full_blk, dict):
            for src, dst in (
                ("host_tail_mean_ms", "full_host_tail_mean_ms"),
                ("host_tail_std_ms", "full_host_tail_std_ms"),
                ("e2e_mean_ms", "full_e2e_mean_ms"),
                ("e2e_std_ms", "full_e2e_std_ms"),
                ("endpoint_mode", "full_endpoint_mode"),
            ):
                if src in full_blk:
                    r.setdefault(dst, full_blk.get(src))
        if isinstance(part2_blk, dict):
            for src, dst in (
                ("accelerator_prefix_mean_ms", "part2_prefix_mean_ms"),
                ("accelerator_prefix_std_ms", "part2_prefix_std_ms"),
                ("host_tail_mean_ms", "part2_host_tail_mean_ms"),
                ("host_tail_std_ms", "part2_host_tail_std_ms"),
                ("endpoint_mode", "part2_endpoint_mode"),
            ):
                if src in part2_blk:
                    r.setdefault(dst, part2_blk.get(src))
        if isinstance(comp_blk, dict):
            pp = comp_blk.get("phase_profile")
            if isinstance(pp, dict):
                for section, prefix in (
                    ("wall", "composed_phase_wall"),
                    ("stage1", "composed_phase_stage1"),
                    ("stage2_input_mapping", "composed_phase_stage2_input_mapping"),
                    ("stage2_prefix", "composed_phase_stage2_prefix"),
                    ("stage2_host_tail", "composed_phase_stage2_host_tail"),
                    ("stage2_total", "composed_phase_stage2_total"),
                ):
                    blk = pp.get(section)
                    if isinstance(blk, dict):
                        if blk.get("mean_ms") is not None:
                            r.setdefault(f"{prefix}_mean_ms", blk.get("mean_ms"))
                        if blk.get("std_ms") is not None:
                            r.setdefault(f"{prefix}_std_ms", blk.get("std_ms"))
                r.setdefault("composed_phase_stage2_uses_hailo_prefix", pp.get("stage2_uses_hailo_prefix"))
                r.setdefault("composed_phase_stage2_uses_host_tail", pp.get("stage2_uses_host_tail"))

    rh = r.get("raw_head_tensor_compare") or (r.get("timing_breakdown") or {}).get("raw_head_drift") if isinstance(r.get("timing_breakdown"), dict) else r.get("raw_head_tensor_compare")
    if isinstance(rh, dict):
        r.setdefault("raw_head_drift_status", rh.get("status"))
        r.setdefault("raw_head_drift_pass", rh.get("pass"))
        r.setdefault("raw_head_drift_max_abs", rh.get("max_abs"))
        r.setdefault("raw_head_drift_mean_abs_mean", rh.get("mean_abs_mean"))
        r.setdefault("raw_head_drift_num_compared_tensors", rh.get("num_compared_tensors"))

    throughput = r.get("throughput")
    if isinstance(throughput, dict):
        for key in (
            "mode",
            "queue_depth",
            "warmup_frames",
            "measured_frames",
            "fps_makespan",
            "fps_cycle_est",
            "cycle_est_ms",
            "makespan_ms",
            "latency_mean_ms",
            "latency_p50_ms",
            "latency_p95_ms",
            "stage1_mean_ms",
            "stage2_mean_ms",
            "frame_count_total",
        ):
            if key in throughput:
                r.setdefault(f"throughput_{key}", throughput.get(key))

    backend_drift = r.get("backend_drift")
    if isinstance(backend_drift, dict):
        r.setdefault("backend_drift_enabled", bool(backend_drift.get("enabled")))
        r.setdefault("backend_drift_status", backend_drift.get("status"))
        r.setdefault("backend_drift_mode", backend_drift.get("mode"))
        r.setdefault("backend_drift_variant", backend_drift.get("variant"))
        r.setdefault("backend_drift_provider", backend_drift.get("provider"))
        r.setdefault("backend_drift_reference_source", backend_drift.get("reference_source"))
        single = backend_drift.get("single_sample") if isinstance(backend_drift.get("single_sample"), dict) else {}
        elem = single.get("elementwise") if isinstance(single.get("elementwise"), dict) else {}
        proxy = single.get("proxy") if isinstance(single.get("proxy"), dict) else {}
        if isinstance(elem, dict):
            r.setdefault("backend_drift_single_max_abs", elem.get("max_abs"))
            r.setdefault("backend_drift_single_mean_abs", elem.get("mean_abs"))
        if isinstance(proxy, dict):
            r.setdefault("backend_drift_single_match_ratio", proxy.get("match_ratio"))
            r.setdefault("backend_drift_single_mean_iou", proxy.get("mean_iou_matched"))
            r.setdefault("backend_drift_single_count_delta_abs", proxy.get("count_delta_abs"))
            r.setdefault("backend_drift_single_pass", proxy.get("pass"))
        cls_proxy = single.get("classification_proxy") if isinstance(single.get("classification_proxy"), dict) else {}
        if isinstance(cls_proxy, dict):
            r.setdefault("backend_drift_single_top1_match", cls_proxy.get("top1_match"))
            r.setdefault("backend_drift_single_top5_agreement", cls_proxy.get("top5_agreement"))
            r.setdefault("backend_drift_single_cosine_similarity", cls_proxy.get("cosine_similarity"))
            r.setdefault("backend_drift_single_mean_abs_cls", cls_proxy.get("mean_abs"))
        dataset = backend_drift.get("dataset") if isinstance(backend_drift.get("dataset"), dict) else {}
        metrics = dataset.get("metrics") if isinstance(dataset.get("metrics"), dict) else {}
        if isinstance(dataset, dict):
            r.setdefault("backend_drift_dataset_pass", dataset.get("pass"))
            r.setdefault("backend_drift_dataset_requested_images", dataset.get("requested_images"))
        if isinstance(metrics, dict):
            r.setdefault("backend_drift_dataset_n_images", metrics.get("n_images"))
            r.setdefault("backend_drift_dataset_image_pass_ratio", metrics.get("image_pass_ratio"))
            r.setdefault("backend_drift_dataset_global_match_ratio", metrics.get("global_match_ratio"))
            r.setdefault("backend_drift_dataset_global_mean_iou", metrics.get("global_mean_iou_matched"))
            r.setdefault("backend_drift_dataset_count_delta_abs", metrics.get("count_delta_abs"))
            r.setdefault("backend_drift_dataset_top1_agreement", metrics.get("top1_agreement"))
            r.setdefault("backend_drift_dataset_top5_agreement", metrics.get("top5_agreement"))
            r.setdefault("backend_drift_dataset_mean_cosine_similarity", metrics.get("mean_cosine_similarity"))
            r.setdefault("backend_drift_dataset_top1_accuracy", metrics.get("top1_accuracy"))
            r.setdefault("backend_drift_dataset_top5_accuracy", metrics.get("top5_accuracy"))

    mini_coco = r.get("mini_coco_ap50")
    if isinstance(mini_coco, dict):
        r.setdefault("mini_coco_ap50_enabled", bool(mini_coco.get("enabled")))
        r.setdefault("mini_coco_ap50_status", mini_coco.get("status"))
        r.setdefault("mini_coco_ap50_reference_source", mini_coco.get("reference_source"))
        variants_blk = mini_coco.get("variants") if isinstance(mini_coco.get("variants"), dict) else {}
        for vv, payload in variants_blk.items():
            if not isinstance(payload, dict):
                continue
            key = str(vv).strip().lower()
            if key:
                r.setdefault(f"mini_coco_ap50_{key}", payload.get("ap50"))
        primary_variant = str(r.get("primary_variant") or "").strip().lower()
        if primary_variant and f"mini_coco_ap50_{primary_variant}" in r:
            r.setdefault("mini_coco_ap50_primary_variant", primary_variant)
            r.setdefault("mini_coco_ap50_primary", r.get(f"mini_coco_ap50_{primary_variant}"))
        if r.get("mini_coco_ap50_primary") is not None and r.get("mini_coco_ap50_full") is not None:
            try:
                r.setdefault("mini_coco_ap50_delta_primary_minus_full", float(r.get("mini_coco_ap50_primary")) - float(r.get("mini_coco_ap50_full")))
            except Exception:
                pass

    mini_cls = r.get("mini_classification_eval")
    if isinstance(mini_cls, dict):
        r.setdefault("mini_classification_eval_enabled", bool(mini_cls.get("enabled")))
        r.setdefault("mini_classification_eval_status", mini_cls.get("status"))
        variants_blk = mini_cls.get("variants") if isinstance(mini_cls.get("variants"), dict) else {}
        for vv, payload in variants_blk.items():
            if not isinstance(payload, dict):
                continue
            key = str(vv).strip().lower()
            if not key:
                continue
            r.setdefault(f"mini_classification_eval_{key}_top1", payload.get("top1_accuracy"))
            r.setdefault(f"mini_classification_eval_{key}_top5", payload.get("top5_accuracy"))
            r.setdefault(f"mini_classification_eval_{key}_images", payload.get("images"))
        primary_variant = str(r.get("primary_variant") or "").strip().lower()
        if primary_variant:
            if f"mini_classification_eval_{primary_variant}_top1" in r:
                r.setdefault("mini_classification_eval_primary_variant", primary_variant)
                r.setdefault("mini_classification_eval_primary_top1", r.get(f"mini_classification_eval_{primary_variant}_top1"))
                r.setdefault("mini_classification_eval_primary_top5", r.get(f"mini_classification_eval_{primary_variant}_top5"))
        if r.get("mini_classification_eval_primary_top1") is not None and r.get("mini_classification_eval_full_top1") is not None:
            try:
                r.setdefault("mini_classification_eval_delta_primary_minus_full_top1", float(r.get("mini_classification_eval_primary_top1")) - float(r.get("mini_classification_eval_full_top1")))
            except Exception:
                pass
        if r.get("mini_classification_eval_primary_top5") is not None and r.get("mini_classification_eval_full_top5") is not None:
            try:
                r.setdefault("mini_classification_eval_delta_primary_minus_full_top5", float(r.get("mini_classification_eval_primary_top5")) - float(r.get("mini_classification_eval_full_top5")))
            except Exception:
                pass

    validation_dataset = r.get("validation_dataset")
    if isinstance(validation_dataset, dict):
        r.setdefault("validation_dataset_task", validation_dataset.get("task"))
        variants_blk = validation_dataset.get("variants") if isinstance(validation_dataset.get("variants"), dict) else {}
        primary_variant = str(r.get("primary_variant") or "").strip().lower()
        primary_payload = variants_blk.get(primary_variant) if primary_variant else None
        if isinstance(primary_payload, dict):
            metrics_blk = primary_payload.get("metrics") if isinstance(primary_payload.get("metrics"), dict) else {}
            r.setdefault("validation_dataset_primary_pass", primary_payload.get("pass"))
            r.setdefault("validation_dataset_primary_image_pass_ratio", metrics_blk.get("image_pass_ratio"))
            r.setdefault("validation_dataset_primary_top1_agreement", metrics_blk.get("top1_agreement"))
            r.setdefault("validation_dataset_primary_top5_agreement", metrics_blk.get("top5_agreement"))
            r.setdefault("validation_dataset_primary_top1_accuracy", metrics_blk.get("top1_accuracy"))
            r.setdefault("validation_dataset_primary_top5_accuracy", metrics_blk.get("top5_accuracy"))

    try:
        p1 = float(r.get("part1_mean_ms", float("nan")))
        p2 = float(r.get("part2_mean_ms", float("nan")))
        comp = float(r.get("composed_mean_ms", float("nan")))
        if math.isfinite(p1) and math.isfinite(p2):
            r["sum_parts_ms"] = p1 + p2
            if math.isfinite(comp):
                r["overhead_ms"] = comp - (p1 + p2)
    except Exception:
        pass
    try:
        full = float(r.get("full_mean_ms", float("nan")))
        comp = float(r.get("composed_mean_ms", float("nan")))
        if math.isfinite(full) and math.isfinite(comp) and comp > 0:
            r["speedup_full_over_composed"] = full / comp
    except Exception:
        pass

    sessions = r.get("sessions")
    if isinstance(sessions, dict):
        runtime_sessions = sessions.get("ort_runtime") if isinstance(sessions.get("ort_runtime"), dict) else {}
        baseline_sessions = sessions.get("baseline_ort_cpu") if isinstance(sessions.get("baseline_ort_cpu"), dict) else {}

        def _pick_session_payload(kind: str, provider_hint: Optional[str]) -> Optional[Dict[str, Any]]:
            hint = str(provider_hint or "").strip().lower()
            candidates: List[str] = []
            if hint:
                candidates.append(f"{kind}:{hint}")
            default_hint = str(provider or "").strip().lower()
            if default_hint and f"{kind}:{default_hint}" not in candidates:
                candidates.append(f"{kind}:{default_hint}")
            for key in candidates:
                payload = runtime_sessions.get(key)
                if isinstance(payload, dict):
                    return payload
            payload = baseline_sessions.get(kind)
            return payload if isinstance(payload, dict) else None

        for kind, hint in (("full", provider), ("part1", stage1 or provider), ("part2", stage2 or provider)):
            payload = _pick_session_payload(kind, hint)
            if not isinstance(payload, dict):
                continue
            build_s = payload.get("build_seconds")
            if build_s is not None:
                r.setdefault(f"{kind}_session_build_s", build_s)
            req = payload.get("providers_requested")
            if isinstance(req, list):
                r.setdefault(f"{kind}_session_providers_requested", "|".join(str(x) for x in req))
            use = payload.get("providers_in_use")
            if isinstance(use, list):
                r.setdefault(f"{kind}_session_providers_in_use", "|".join(str(x) for x in use))

    r.setdefault("provider", provider)
    r.setdefault("image_scale", image_scale)
    if "benchmark_task_used" in r:
        r.setdefault("benchmark_task", r.get("benchmark_task_used"))
    elif "benchmark_task_requested" in r:
        r.setdefault("benchmark_task", r.get("benchmark_task_requested"))
    return r


def _path_looks_like_semantic_validation_source(path: Path) -> bool:
    image_exts = {'.jpg', '.jpeg', '.png', '.bmp', '.webp'}
    list_exts = {'.txt', '.json'}
    try:
        if path.is_file():
            return path.suffix.lower() in image_exts | list_exts
        if path.is_dir():
            for cand in path.rglob('*'):
                if cand.is_file() and cand.suffix.lower() in image_exts:
                    return True
            for cand in path.iterdir():
                if cand.is_file() and cand.suffix.lower() in list_exts:
                    return True
    except Exception:
        return False
    return False



def _run_case(case_dir: Path, run_id: str, provider: str, image: str, preset: str, image_scale: str, warmup: int, runs: int, timeout_s: Optional[int], stage1: Optional[str] = None, stage2: Optional[str] = None, variants: Optional[List[str]] = None, case_meta: Optional[Dict[str, Any]] = None, *, resume: bool = False, verbose_runs: bool = False, progress_every: int = 25, trt_cache_root: Optional[str] = None, throughput_frames: int = 24, throughput_warmup_frames: int = 6, throughput_queue_depth: int = 2, phase_runs: int = 5, validation_images: Optional[str] = None, validation_max_images: int = 0, validation_reference_mode: str = "auto", mini_coco_ap50: bool = False, benchmark_task: str = "auto", mini_classification_eval: bool = False) -> Optional[Dict[str, Any]]:
    # Each case contains run_split_onnxruntime.py
    runner = case_dir / "run_split_onnxruntime.py"
    if not runner.exists():
        print(f"[warn] missing runner: {runner}")
        return None
    try:
        _runner_probe = runner.read_text(encoding="utf-8", errors="ignore")[:50000]
        runner_supports_phase_runs = "--phase-runs" in _runner_probe
    except Exception:
        runner_supports_phase_runs = False

    out_tag = _slug(run_id)
    out_dir_name = f"results_{out_tag}"

    if resume:
        existing = _collect_case_result(
            case_dir,
            out_dir_name,
            run_id,
            provider,
            image_scale,
            stage1=stage1,
            stage2=stage2,
            variants=variants,
            case_meta=case_meta,
            runner_rc=0,
        )
        if existing is not None:
            existing.setdefault("_resumed", True)
            print(f"[resume] {case_dir.name}: using existing results from {out_dir_name}/validation_report.json")
            return existing

    cmd = [
        sys.executable,
        str(runner),
        "--provider",
        provider,
        "--image-scale",
        image_scale,
        "--warmup",
        str(warmup),
        "--runs",
        str(runs),
        "--out-dir",
        out_dir_name,
        "--throughput-frames",
        str(max(0, int(throughput_frames))),
        "--throughput-warmup-frames",
        str(max(0, int(throughput_warmup_frames))),
        "--throughput-queue-depth",
        str(max(1, int(throughput_queue_depth))),
    ]
    if runner_supports_phase_runs:
        cmd += ["--phase-runs", str(max(0, int(phase_runs)))]
    elif int(phase_runs or 0) > 0:
        print(f"[warn] runner does not support --phase-runs yet: {runner.name}; v42 phase diagnostics will be skipped for this case")
    if variants:
        cmd += ["--variants", ",".join([str(v) for v in variants])]
    if verbose_runs:
        cmd += ["--verbose-runs"]
    if stage1:
        cmd += ["--stage1-provider", str(stage1)]
    if stage2:
        cmd += ["--stage2-provider", str(stage2)]
    task_norm = str(benchmark_task or "auto").strip().lower() or "auto"
    val_img: Optional[Path] = None
    default_val = (case_dir.parent / "resources" / "validation" / "coco_50_data").resolve()
    if validation_images:
        val_img = Path(str(validation_images))
        if not val_img.is_absolute():
            # benchmark_plan.json lives at suite root; resolve relative semantic
            # validation paths against the suite root, not the case directory.
            val_img = (case_dir.parent / val_img).resolve()
        if not _path_looks_like_semantic_validation_source(val_img):
            if task_norm == "classification":
                print(f"[warn] validation_images={val_img} does not look like a labeled classification validation source; disabling dataset validation")
                val_img = None
                validation_max_images = 0
            else:
                print(f"[warn] validation_images={val_img} does not look like a semantic validation source; falling back to prepared COCO-50")
                val_img = default_val if default_val.exists() else None
                if val_img is not None and int(validation_max_images or 0) <= 0:
                    validation_max_images = 50
    else:
        if task_norm != "classification" and default_val.exists():
            val_img = default_val
            if int(validation_max_images or 0) <= 0:
                validation_max_images = 50
    if val_img is not None:
        cmd += ["--validation-images", str(val_img)]
    if int(validation_max_images or 0) > 0:
        cmd += ["--validation-max-images", str(int(validation_max_images))]
    if str(validation_reference_mode or "").strip():
        cmd += ["--validation-reference-mode", str(validation_reference_mode)]
    if str(benchmark_task or "").strip():
        cmd += ["--benchmark-task", str(benchmark_task)]
    if bool(mini_coco_ap50):
        cmd += ["--mini-coco-ap50"]
    if bool(mini_classification_eval):
        cmd += ["--mini-classification-eval"]

    def _uses_tensorrt_token(token: Optional[str]) -> bool:
        return str(token or "").strip().lower() in {"tensorrt", "trt"}

    if trt_cache_root and any(_uses_tensorrt_token(tok) for tok in (provider, stage1, stage2)):
        cache_root = Path(str(trt_cache_root))
        if not cache_root.is_absolute():
            cache_root = (case_dir.parent / cache_root).resolve()
        cache_dir = cache_root / case_dir.name
        cache_dir.mkdir(parents=True, exist_ok=True)
        cmd += ["--trt-cache-dir", str(cache_dir)]

    if image and str(image).lower() != "default":
        cmd += ["--image", image]
    rc = 0
    try:
        res = subprocess.run(cmd, cwd=str(case_dir), timeout=timeout_s)
        rc = int(getattr(res, "returncode", 0) or 0)
        if rc != 0:
            print(f"[warn] case failed: {case_dir.name} (rc={rc})")
    except subprocess.TimeoutExpired:
        print(f"[warn] timeout: {case_dir.name} (>{timeout_s}s)")
        return None

    return _collect_case_result(
        case_dir,
        out_dir_name,
        run_id,
        provider,
        image_scale,
        stage1=stage1,
        stage2=stage2,
        variants=variants,
        case_meta=case_meta,
        runner_rc=rc,
    )

def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--provider", default="auto", choices=["auto", "cpu", "cuda", "tensorrt"], help="Execution provider to use (single-run mode).")
    parser.add_argument(
        "--plan",
        default=None,
        help="Optional benchmark plan JSON (default: benchmark_plan.json if present). If set, runs all runs listed in the plan.",
    )
    parser.add_argument("--run-id", default=None, help="If using --plan, run only this run id.")
    parser.add_argument("--list-runs", action="store_true", help="List run ids from the plan and exit.")
    parser.add_argument(
        "--image",
        default="default",
        help=(
            "Image for CV inputs. Either a path or one of: 'default'|'coco'|'imagenet'. "
            "(Passed through to per-case runners.)"
        ),
    )
    parser.add_argument("--preset", default="auto", choices=["auto", "classification", "detection"], help="Output agreement preset.")
    parser.add_argument(
        "--image-scale",
        default="auto",
        choices=["auto", "norm", "raw", "imagenet", "clip"],
        help="Input image scaling mode for the runner. In plan-mode, runs can override this via run.image_scale.",
    )
    parser.add_argument("--validation-images", default="", help="Optional semantic-validation dataset source (detection image list / classification labeled dataset). In plan-mode, runs can override this via run.validation_images.")
    parser.add_argument("--validation-max-images", type=int, default=0, help="Max samples from --validation-images to use per case (0 disables dataset semantic validation for classification; detection falls back to prepared COCO-50 when appropriate).")
    parser.add_argument("--validation-reference-mode", default="auto", choices=["auto", "same_backend_full", "cpu_full", "annotations"], help="Reference policy for semantic validation. In plan-mode, runs can override this via run.validation_reference_mode.")
    parser.add_argument("--benchmark-task", default="auto", choices=["auto", "detection", "classification"], help="Task-aware validation/reporting mode. In plan-mode, runs can override this via run.benchmark_task.")
    parser.add_argument("--mini-coco-ap50", action="store_true", help="Optional report-only Mini-COCO AP50 on the semantic validation set for detection tasks. In plan-mode, runs can override this via run.mini_coco_ap50.")
    parser.add_argument("--mini-classification-eval", action="store_true", help="Optional report-only Top-1/Top-5 classification summary on the semantic validation set. In plan-mode, runs can override this via run.mini_classification_eval.")
    parser.add_argument("--warmup", type=int, default=5)
    parser.add_argument("--runs", type=int, default=10)
    parser.add_argument("--phase-runs", type=int, default=5, help="Forward v42 diagnostic phase timing runs to per-case runners (0 disables).")
    parser.add_argument("--throughput-frames", type=int, default=24, help="Measured streaming/interleaving frames for heterogeneous stage1→stage2 pipelines (0 disables).")
    parser.add_argument("--throughput-warmup-frames", type=int, default=6, help="Warmup frames before measured streaming/interleaving throughput.")
    parser.add_argument("--throughput-queue-depth", type=int, default=2, help="Queue depth for the streaming/interleaving throughput probe.")
    parser.add_argument("--verbose-runs", action="store_true", help="Forward --verbose-runs to the per-case runner.")
    parser.add_argument("--progress-every", type=int, default=25, help="Forward sparse progress reporting to the per-case runner (0 disables it).")
    parser.add_argument("--timeout", type=int, default=None, help="Per-case timeout in seconds.")
    parser.add_argument("--resume", action="store_true", help="Reuse existing per-case results_<run_id>/validation_report.json files and skip already finished cases.")
    parser.add_argument("--trt-cache-root", default=None, help="Optional shared TensorRT cache root. When set, each case uses <root>/<case_dir>/ as --trt-cache-dir so TRT engines/timing caches can be reused across suite reruns.")
    parser.add_argument("--objective", default="composed", choices=["composed", "sum_parts", "max_parts", "full"], help="Objective for ranking/summaries.")
    parser.add_argument("--topk", type=int, default=10, help="Top-k for highlights/tables.")
    parser.add_argument("--no-csv", action="store_true", help="Do not export CSV.")
    parser.add_argument(
        "--no-plot",
        action="store_true",
        help="Disable plot generation (paper figures / overview plots).",
    )
    args = parser.parse_args()

    root = Path(__file__).resolve().parent
    bench = _read_json(root / "__BENCH_JSON__")
    cases: List[Dict[str, Any]] = bench.get("cases", [])
    if not cases:
        print("[warn] __BENCH_JSON__ has no cases")
        return 1
    suite_hailo_hefs = ((bench.get("hailo") or {}).get("hefs") if isinstance(bench.get("hailo"), dict) else None)
    suite_hailo_hefs = suite_hailo_hefs if isinstance(suite_hailo_hefs, dict) else {}

    # Decide between plan-mode and single provider mode.
    plan_path: Optional[Path] = None
    if args.plan:
        plan_path = (root / args.plan).resolve() if not os.path.isabs(args.plan) else Path(args.plan)
    else:
        # Auto-enable plan mode only if the user did not explicitly request a provider.
        # (provider='auto' is the default)
        default_plan = root / "benchmark_plan.json"
        if args.provider == "auto" and default_plan.exists():
            plan_path = default_plan

    plan: Optional[Dict[str, Any]] = None
    runs_plan: List[Dict[str, Any]] = []
    if plan_path is not None and plan_path.exists():
        try:
            plan = _read_json(plan_path)
            if isinstance(plan, dict) and isinstance(plan.get("runs"), list):
                runs_plan = [r for r in plan.get("runs") if isinstance(r, dict)]
        except Exception as e:
            print(f"[warn] Failed to read plan file {plan_path}: {type(e).__name__}: {e}")
            plan = None
            runs_plan = []

    if args.list_runs:
        if not runs_plan:
            print("(no plan / no runs)")
            return 0
        for r in runs_plan:
            rid = r.get("id") or r.get("name") or "(unnamed)"
            rty = r.get("type") or r.get("kind") or "?"
            print(f"{rid}\t{rty}")
        return 0

    if runs_plan:
        # Optional filter
        if args.run_id:
            runs_plan = [r for r in runs_plan if str(r.get("id") or r.get("name") or "").strip() == str(args.run_id).strip()]
            if not runs_plan:
                print(f"[err] run id not found in plan: {args.run_id}")
                return 2
        runs = runs_plan
    else:
        # Fallback: single provider run
        runs = [{"id": f"ort_{args.provider}", "type": "onnxruntime", "provider": args.provider, "image_scale": args.image_scale}]

    # Execute each run.
    # Robust exit code policy:
    # - If at least one configured run produced results, return rc=0 (success)
    #   and write a status file listing runs with no results.
    # - If *no* run produced results, return rc=2.
    overall_rc = 0
    any_rows = False
    failed_runs: List[Dict[str, Any]] = []
    rows_by_tag: Dict[str, List[Dict[str, Any]]] = {}
    for run in runs:
        rtype = str(run.get("type") or run.get("kind") or "onnxruntime").strip().lower()
        run_id = str(run.get("id") or run.get("name") or rtype).strip() or rtype

        if rtype in {"onnxruntime", "ort"}:
            provider = str(run.get("provider") or args.provider).strip().lower()
            if provider not in {"auto", "cpu", "cuda", "tensorrt"}:
                print(f"[warn] Unknown provider '{provider}' for run {run_id}; skipping")
                continue

            # Optional stage config / variants (used for matrix-style runs).
            stage1_tok = _resolve_stage_token(run.get("stage1")) or provider
            stage2_tok = _resolve_stage_token(run.get("stage2")) or provider
            variants = _normalize_variants(run.get("variants"))

            tag = f"{run_id}_{args.preset}"
            rows: List[Dict[str, Any]] = []
            for i, c in enumerate(cases):
                b = c.get("boundary")
                case_dir_key = c.get("case_dir") or c.get("folder")
                if case_dir_key is None:
                    case_dir_key = f"b{int(b):03d}" if b is not None else "case"
                rel = Path(str(case_dir_key))
                case_dir = (root / rel).resolve()
                variants_eff = _prune_case_variants_for_run(run, variants, c, suite_hailo_hefs)
                if variants is not None and not variants_eff:
                    print(f"\n[{run_id}] [{i+1}/{len(cases)}] Skipping {case_dir.name} (no usable variants after Hailo availability pruning)")
                    continue

                print(f"\n[{run_id}] [{i+1}/{len(cases)}] Running {case_dir.name} (provider={provider})")

                # Per-run image scaling (passed through to run_split_onnxruntime.py).
                image_scale = str(run.get("image_scale") or getattr(args, "image_scale", "auto")).strip().lower()
                if image_scale not in ("auto", "norm", "raw", "imagenet", "clip"):
                    print(f"[warn] invalid image_scale={image_scale!r} in plan for run {run_id!r}; falling back to 'auto'")
                    image_scale = "auto"
                cli_validation_images = str(getattr(args, "validation_images", "") or "").strip()
                if cli_validation_images:
                    validation_images = cli_validation_images
                    validation_max_images = int(getattr(args, "validation_max_images", 0) or 0)
                else:
                    validation_images = str(run.get("validation_images") or "").strip()
                    validation_max_images = int(run.get("validation_max_images") or getattr(args, "validation_max_images", 0) or 0)
                validation_reference_mode = str(run.get("validation_reference_mode") or getattr(args, "validation_reference_mode", "auto") or "auto").strip().lower()
                if validation_reference_mode not in ("auto", "same_backend_full", "cpu_full", "annotations"):
                    validation_reference_mode = "auto"
                benchmark_task = str(run.get("benchmark_task") or getattr(args, "benchmark_task", "auto") or "auto").strip().lower()
                if benchmark_task not in ("auto", "detection", "classification"):
                    benchmark_task = "auto"
                mini_coco_ap50 = bool(run.get("mini_coco_ap50") if "mini_coco_ap50" in run else getattr(args, "mini_coco_ap50", False))
                mini_classification_eval = bool(run.get("mini_classification_eval") if "mini_classification_eval" in run else getattr(args, "mini_classification_eval", False))

                r = _run_case(
                    case_dir,
                    run_id=run_id,
                    provider=provider,
                    stage1=stage1_tok,
                    stage2=stage2_tok,
                    variants=variants_eff,
                    image=args.image,
                    preset=args.preset,
                    image_scale=image_scale,
                    warmup=args.warmup,
                    runs=args.runs,
                    phase_runs=args.phase_runs,
                    timeout_s=args.timeout,
                    case_meta=c,
                    resume=getattr(args, "resume", False),
                    verbose_runs=args.verbose_runs,
                    progress_every=args.progress_every,
                    trt_cache_root=args.trt_cache_root,
                    throughput_frames=args.throughput_frames,
                    throughput_warmup_frames=args.throughput_warmup_frames,
                    throughput_queue_depth=args.throughput_queue_depth,
                    validation_images=validation_images,
                    validation_max_images=validation_max_images,
                    validation_reference_mode=validation_reference_mode,
                    mini_coco_ap50=mini_coco_ap50,
                    benchmark_task=benchmark_task,
                    mini_classification_eval=mini_classification_eval,
                )
                if r is not None:
                    rows.append(r)

            if not rows:
                print(f"[warn] No results collected for run {run_id}")
                failed_runs.append({"run_id": run_id, "type": rtype, "provider": provider})

                # Still write placeholder artifacts so UIs/parsers can show the run as "empty".
                results_json = root / f"benchmark_results_{tag}.json"
                _write_json(results_json, rows)
                rows_by_tag[str(tag)] = list(rows)
                try:
                    (root / f"benchmark_results_{tag}.error.txt").write_text(
                        f"No results collected for run {run_id} (provider={provider}).\n",
                        encoding="utf-8",
                    )
                except Exception:
                    pass
                continue

            any_rows = True

            results_json = root / f"benchmark_results_{tag}.json"
            _write_json(results_json, rows)
            rows_by_tag[str(tag)] = list(rows)
            print(f"Wrote {results_json.name}")

            if not args.no_csv:
                results_csv = root / f"benchmark_results_{tag}.csv"
                _write_results_csv(results_csv, rows)
                print(f"Wrote {results_csv.name}")
            # Summary outputs are best-effort: benchmarking should not fail if plotting/formatting fails.

            # Summary (1 page)
            summary_md = root / f"benchmark_summary_{tag}.md"
            try:
                _write_summary_md(summary_md, rows, tag=tag, objective=args.objective, topk=args.topk)
                print(f"Wrote {summary_md.name}")
            except Exception as e:
                print(f"[warn] Failed to write {summary_md.name}: {e}")

            # Paper-ready LaTeX table
            table_tex = root / f"benchmark_table_{tag}.tex"
            try:
                _write_table_tex(table_tex, rows, tag=tag, objective=args.objective, topk=args.topk)
                print(f"Wrote {table_tex.name}")
            except Exception as e:
                print(f"[warn] Failed to write {table_tex.name}: {e}")

            # Paper-ready figures
            fig_dir = root / f"paper_figures_{tag}"
            try:
                fig_files = _save_paper_figures(fig_dir, rows, tag=tag, objective=args.objective, topk=args.topk)
                if fig_files:
                    print(f"Wrote {len(fig_files)} figure files to {fig_dir.name}/")
                else:
                    if not HAVE_PLOT:
                        print("[info] matplotlib not available; skipping figure export")
                    else:
                        print("[info] no figures exported (no rows?)")
            except Exception as e:
                print(f"[warn] Failed to generate figures in {fig_dir.name}/: {e}")


        elif rtype in {"matrix"}:
            # Matrix run: stage1/stage2 can be on different backends (ORT vs Hailo, etc.)
            stage1_tok = _resolve_stage_token(run.get("stage1"))
            stage2_tok = _resolve_stage_token(run.get("stage2"))
            if not stage1_tok or not stage2_tok:
                print(f"[warn] matrix run {run_id!r} is missing stage1/stage2; skipping")
                continue

            # Default provider passed to the per-case runner (acts as a fallback). Use stage1.
            provider_tag = str(run.get("provider") or stage1_tok).strip().lower() or stage1_tok

            variants = _normalize_variants(run.get("variants"))

            tag = f"{run_id}_{args.preset}"
            rows: List[Dict[str, Any]] = []
            for i, c in enumerate(cases):
                b = c.get("boundary")
                case_dir_key = c.get("case_dir") or c.get("folder")
                if case_dir_key is None:
                    case_dir_key = f"b{int(b):03d}" if b is not None else "case"
                rel = Path(str(case_dir_key))
                case_dir = (root / rel).resolve()
                variants_eff = _prune_case_variants_for_run(run, variants, c, suite_hailo_hefs)
                if variants is not None and not variants_eff:
                    print(
                        f"\n[{run_id}] [{i+1}/{len(cases)}] Skipping {case_dir.name} "
                        f"(no usable variants after Hailo availability pruning)"
                    )
                    continue

                print(
                    f"\n[{run_id}] [{i+1}/{len(cases)}] Running {case_dir.name} "
                    f"(stage1={stage1_tok}, stage2={stage2_tok})"
                )

                # Per-run image scaling (passed through to run_split_onnxruntime.py).
                image_scale = str(run.get("image_scale") or getattr(args, "image_scale", "auto")).strip().lower()
                if image_scale not in ("auto", "norm", "raw", "imagenet", "clip"):
                    print(f"[warn] invalid image_scale={image_scale!r} in plan for run {run_id!r}; falling back to 'auto'")
                    image_scale = "auto"
                cli_validation_images = str(getattr(args, "validation_images", "") or "").strip()
                if cli_validation_images:
                    validation_images = cli_validation_images
                    validation_max_images = int(getattr(args, "validation_max_images", 0) or 0)
                else:
                    validation_images = str(run.get("validation_images") or "").strip()
                    validation_max_images = int(run.get("validation_max_images") or getattr(args, "validation_max_images", 0) or 0)
                validation_reference_mode = str(run.get("validation_reference_mode") or getattr(args, "validation_reference_mode", "auto") or "auto").strip().lower()
                if validation_reference_mode not in ("auto", "same_backend_full", "cpu_full", "annotations"):
                    validation_reference_mode = "auto"
                benchmark_task = str(run.get("benchmark_task") or getattr(args, "benchmark_task", "auto") or "auto").strip().lower()
                if benchmark_task not in ("auto", "detection", "classification"):
                    benchmark_task = "auto"
                mini_coco_ap50 = bool(run.get("mini_coco_ap50") if "mini_coco_ap50" in run else getattr(args, "mini_coco_ap50", False))
                mini_classification_eval = bool(run.get("mini_classification_eval") if "mini_classification_eval" in run else getattr(args, "mini_classification_eval", False))

                r = _run_case(
                    case_dir,
                    run_id=run_id,
                    provider=provider_tag,
                    stage1=stage1_tok,
                    stage2=stage2_tok,
                    variants=variants_eff,
                    image=args.image,
                    preset=args.preset,
                    image_scale=image_scale,
                    warmup=args.warmup,
                    runs=args.runs,
                    phase_runs=args.phase_runs,
                    timeout_s=args.timeout,
                    case_meta=c,
                    resume=getattr(args, "resume", False),
                    verbose_runs=args.verbose_runs,
                    progress_every=args.progress_every,
                    trt_cache_root=args.trt_cache_root,
                    throughput_frames=args.throughput_frames,
                    throughput_warmup_frames=args.throughput_warmup_frames,
                    throughput_queue_depth=args.throughput_queue_depth,
                    validation_images=validation_images,
                    validation_max_images=validation_max_images,
                    validation_reference_mode=validation_reference_mode,
                    mini_coco_ap50=mini_coco_ap50,
                    benchmark_task=benchmark_task,
                    mini_classification_eval=mini_classification_eval,
                )
                if r is not None:
                    rows.append(r)

            if not rows:
                print(f"[warn] No results collected for run {run_id}")
                failed_runs.append({"run_id": run_id, "type": rtype, "provider": provider_tag, "stage1": stage1_tok, "stage2": stage2_tok})

                # Still write placeholder artifacts so UIs/parsers can show the run as "empty".
                results_json = root / f"benchmark_results_{tag}.json"
                _write_json(results_json, rows)
                rows_by_tag[str(tag)] = list(rows)
                try:
                    (root / f"benchmark_results_{tag}.error.txt").write_text(
                        f"No results collected for run {run_id} (stage1={stage1_tok}, stage2={stage2_tok}).\n",
                        encoding="utf-8",
                    )
                except Exception:
                    pass
                continue

            any_rows = True

            results_json = root / f"benchmark_results_{tag}.json"
            _write_json(results_json, rows)
            rows_by_tag[str(tag)] = list(rows)
            print(f"Wrote {results_json.name}")

            if not args.no_csv:
                results_csv = root / f"benchmark_results_{tag}.csv"
                _write_results_csv(results_csv, rows)
                print(f"Wrote {results_csv.name}")

            # Summary outputs are best-effort: benchmarking should not fail if plotting/formatting fails.

            summary_md = root / f"benchmark_summary_{tag}.md"
            try:
                _write_summary_md(summary_md, rows, tag=tag, objective=args.objective, topk=args.topk)
                print(f"Wrote {summary_md.name}")
            except Exception as e:
                print(f"[warn] Failed to write summary md: {e}")

            table_tex = root / f"benchmark_table_{tag}.tex"
            try:
                _write_table_tex(table_tex, rows, tag=tag, objective=args.objective, topk=args.topk)
                print(f"Wrote {table_tex.name}")
            except Exception as e:
                print(f"[warn] Failed to write TeX table: {e}")

            # Paper-ready figures (plots)
            # NOTE: args.no_plot was added after some suites were already generated.
            # Using getattr keeps older bundles runnable even if the flag is missing.
            if HAVE_PLOT and not getattr(args, "no_plot", False):
                fig_dir = root / f"paper_figures_{tag}"
                try:
                    _write_figures(fig_dir, rows, tag=tag, objective=args.objective)
                    print(f"Wrote figures in {fig_dir.name}/")
                except Exception as e:
                    print(f"[warn] Failed to generate figures in {fig_dir.name}/: {e}")

        elif rtype in {"hailo"}:
            # Hailo benchmark: reuse the per-case runner script, but select a provider tag
            # that the runner interprets as a Hailo backend (e.g. "hailo8", "hailo8l",
            # "hailo10h", "hailo10p").
            hw_arch = str(run.get("hw_arch") or run.get("arch") or run_id or "").strip()
            provider_tag = hw_arch or "hailo"

            stage1_tok = _resolve_stage_token(run.get("stage1")) or provider_tag
            stage2_tok = _resolve_stage_token(run.get("stage2")) or provider_tag
            variants = _normalize_variants(run.get("variants"))

            tag = f"{run_id}_{args.preset}"
            rows: List[Dict[str, Any]] = []
            for i, c in enumerate(cases):
                b = c.get("boundary")
                case_dir_key = c.get("case_dir") or c.get("folder")
                if case_dir_key is None:
                    case_dir_key = f"b{int(b):03d}" if b is not None else "case"
                rel = Path(str(case_dir_key))
                case_dir = (root / rel).resolve()
                variants_eff = _prune_case_variants_for_run(run, variants, c, suite_hailo_hefs)
                if variants is not None and not variants_eff:
                    print(f"\n[{run_id}] [{i+1}/{len(cases)}] Skipping {case_dir.name} (no usable variants after Hailo availability pruning)")
                    continue

                print(f"\n[{run_id}] [{i+1}/{len(cases)}] Running {case_dir.name} (provider={provider_tag})")

                # Per-run image scaling (passed through to run_split_onnxruntime.py).
                image_scale = str(run.get("image_scale") or getattr(args, "image_scale", "auto")).strip().lower()
                if image_scale not in ("auto", "norm", "raw", "imagenet", "clip"):
                    print(f"[warn] invalid image_scale={image_scale!r} in plan for run {run_id!r}; falling back to 'auto'")
                    image_scale = "auto"
                cli_validation_images = str(getattr(args, "validation_images", "") or "").strip()
                if cli_validation_images:
                    validation_images = cli_validation_images
                    validation_max_images = int(getattr(args, "validation_max_images", 0) or 0)
                else:
                    validation_images = str(run.get("validation_images") or "").strip()
                    validation_max_images = int(run.get("validation_max_images") or getattr(args, "validation_max_images", 0) or 0)
                validation_reference_mode = str(run.get("validation_reference_mode") or getattr(args, "validation_reference_mode", "auto") or "auto").strip().lower()
                if validation_reference_mode not in ("auto", "same_backend_full", "cpu_full", "annotations"):
                    validation_reference_mode = "auto"
                benchmark_task = str(run.get("benchmark_task") or getattr(args, "benchmark_task", "auto") or "auto").strip().lower()
                if benchmark_task not in ("auto", "detection", "classification"):
                    benchmark_task = "auto"
                mini_coco_ap50 = bool(run.get("mini_coco_ap50") if "mini_coco_ap50" in run else getattr(args, "mini_coco_ap50", False))
                mini_classification_eval = bool(run.get("mini_classification_eval") if "mini_classification_eval" in run else getattr(args, "mini_classification_eval", False))

                r = _run_case(
                    case_dir,
                    run_id=run_id,
                    provider=provider_tag,
                    stage1=stage1_tok,
                    stage2=stage2_tok,
                    variants=variants_eff,
                    image=args.image,
                    preset=args.preset,
                    image_scale=image_scale,
                    warmup=args.warmup,
                    runs=args.runs,
                    phase_runs=args.phase_runs,
                    timeout_s=args.timeout,
                    case_meta=c,
                    resume=getattr(args, "resume", False),
                    verbose_runs=args.verbose_runs,
                    progress_every=args.progress_every,
                    trt_cache_root=args.trt_cache_root,
                    throughput_frames=args.throughput_frames,
                    throughput_warmup_frames=args.throughput_warmup_frames,
                    throughput_queue_depth=args.throughput_queue_depth,
                    validation_images=validation_images,
                    validation_max_images=validation_max_images,
                    validation_reference_mode=validation_reference_mode,
                    mini_coco_ap50=mini_coco_ap50,
                    benchmark_task=benchmark_task,
                    mini_classification_eval=mini_classification_eval,
                )
                if r is not None:
                    rows.append(r)

            if not rows:
                print(f"[warn] No results collected for run {run_id}")
                failed_runs.append({"run_id": run_id, "type": rtype, "provider": provider_tag})

                # Still write placeholder artifacts so UIs/parsers can show the run as "empty".
                results_json = root / f"benchmark_results_{tag}.json"
                _write_json(results_json, rows)
                rows_by_tag[str(tag)] = list(rows)
                try:
                    (root / f"benchmark_results_{tag}.error.txt").write_text(
                        f"No results collected for run {run_id} (provider={provider_tag}).\n",
                        encoding="utf-8",
                    )
                except Exception:
                    pass
                continue

            any_rows = True

            results_json = root / f"benchmark_results_{tag}.json"
            _write_json(results_json, rows)
            rows_by_tag[str(tag)] = list(rows)
            print(f"Wrote {results_json.name}")

            if not args.no_csv:
                results_csv = root / f"benchmark_results_{tag}.csv"
                _write_results_csv(results_csv, rows)
                print(f"Wrote {results_csv.name}")

            # Summary outputs are best-effort: benchmarking should not fail if plotting/formatting fails.
            summary_md = root / f"benchmark_summary_{tag}.md"
            try:
                _write_summary_md(summary_md, rows, tag=tag, objective=args.objective, topk=args.topk)
                print(f"Wrote {summary_md.name}")
            except Exception as e:
                print(f"[warn] Failed to write {summary_md.name}: {e}")

            table_tex = root / f"benchmark_table_{tag}.tex"
            try:
                _write_table_tex(table_tex, rows, tag=tag, objective=args.objective, topk=args.topk)
                print(f"Wrote {table_tex.name}")
            except Exception as e:
                print(f"[warn] Failed to write {table_tex.name}: {e}")

            fig_dir = root / f"paper_figures_{tag}"
            try:
                fig_files = _save_paper_figures(fig_dir, rows, tag=tag, objective=args.objective, topk=args.topk)
                if fig_files:
                    print(f"Wrote {len(fig_files)} figure files to {fig_dir.name}/")
                else:
                    if not HAVE_PLOT:
                        print("[info] matplotlib not available; skipping figure export")
                    else:
                        print("[info] no figures exported (no rows?)")
            except Exception as e:
                print(f"[warn] Failed to generate figures in {fig_dir.name}/: {e}")
        else:
            print(f"[warn] Unknown run type '{rtype}' for run {run_id}; skipping")
            continue

    if rows_by_tag:
        try:
            _write_v42_pipeline_summary(root, rows_by_tag)
            print("Wrote v42_pipeline_summary.json/csv/md")
            print("Wrote v47_pipeline_summary.json/csv/md")
        except Exception as e:
            print(f"[warn] Failed to write v42_pipeline_summary: {e}")

    # Always write a small status file so UIs can show partial success.
    try:
        status = {
            "any_rows": any_rows,
            "failed_runs": failed_runs,
            "total_runs": len(runs),
            "total_cases": len(cases),
        }
        status_path = root / "benchmark_suite_status.json"
        _write_json(status_path, status)
        try:
            _write_status_matrix(root, runs)
            print("Wrote benchmark_suite_status_matrix.md/csv/json")
        except Exception as e:
            print(f"[warn] Failed to write benchmark_suite_status_matrix: {e}")
        if failed_runs:
            print(f"[warn] Some runs produced no results; see {status_path.name}")
    except Exception as e:
        print(f"[warn] Failed to write benchmark_suite_status.json: {e}")

    print("\nDone.")

    # Final exit code.
    if not any_rows:
        overall_rc = 2
    else:
        overall_rc = 0
    return overall_rc


if __name__ == "__main__":
    raise SystemExit(main())
